61 template <typename Type, size_t SIZE>
62 void SingletypeThreadsafeObject<Type, SIZE>::set(const Type& datum,
63  const size_t& index)
64 {
65  // we sleep for a nanosecond, in case some thread calls set several
66  // times in a row. this way we do hopefully not miss messages
67  real_time_tools::Timer::sleep_sec(1e-9);
68  // set datum in our data_ member ---------------------------------------
69  {
70  std::unique_lock<std::mutex> lock((*data_mutexes_)[index]);

77  (*modification_counts_)[index] += 1;
